\section{Definitions}

\begin{definition}[Weyl dimension for SU(3)]
For $(m,n) \in \mathbb{N} \times \mathbb{N}$,
$$
\dim(m,n) := \frac{(m+1)(n+1)(m+n+2)}{2}.
$$
\end{definition}

\begin{definition}[Quadratic Casimir]
$$
C_2(m,n) := \frac{1}{3}\left(m^2 + n^2 + mn + 3m + 3n\right).
$$
\end{definition}

\begin{definition}[Truncated Weyl sum]
For $t > 0$, $N \in \mathbb{N}$,
$$
S_N(t) := \sum_{m+n \leq N} \dim(m,n)^2 e^{-t C_2(m,n)}.
$$
\end{definition}

\begin{definition}[Heat kernel at identity]
$$
K(t) := \sum_{(m,n) \in \mathbb{N}^2} \dim(m,n)^2 e^{-t C_2(m,n)}.
$$
\end{definition}

\section{Main Theorems}

\begin{theorem}[Peter-Weyl Summability]
\label{thm:summable}
For all $t > 0$,
$$
\sum_{(m,n)} \dim(m,n)^2 e^{-t C_2(m,n)} < \infty.
$$
\end{theorem}

\begin{proof}
Use $\dim(m,n) \leq C (m+n+1)^2$ and $C_2(m,n) \geq c (m^2+n^2)$. Then 
$$
\sum_{m,n} \dim(m,n)^2 e^{-t C_2(m,n)} \leq C' \sum_{k \geq 0} k^5 e^{-c t k^2} < \infty.
$$
Formalized as \texttt{PeterWeyl\_Summable\_SU3}.
\end{proof}

\begin{theorem}[Finite to infinite bound]
\label{thm:bridge}
For all $t > 0$, $N \in \mathbb{N}$,
$$
S_N(t) \leq K(t).
$$
\end{theorem}

\begin{proof}
Nonnegative terms and Theorem \ref{thm:summable} imply partial sums are bounded by the series. Uses \texttt{Summable.sum\_le\_tsum}.
\end{proof}

\begin{theorem}[Heat kernel asymptotics]
\label{thm:asymp}
There exist $C,c>0$ such that for all $\beta > 0$,
$$
K(1/\beta) \leq C \beta^{-4} e^{-c/\beta}.
$$
\end{theorem}

\begin{proof}
Standard small-$t$ asymptotics for heat kernels on compact Lie groups. For SU(3), $\dim G = 8$ yields exponent $-4$.
\end{proof}

\begin{corollary}[Plaquette activity bound]
\label{cor:main1}
The plaquette activity satisfies
$$
\texttt{plaquette\_activity\_pw}(\beta, N, p) \leq C \beta^{-4} e^{-c/\beta}.
$$
\end{corollary}

\begin{proof}
$\texttt{plaquette\_activity\_pw}$ is defined via $S_N(1/\beta)$. Apply Theorems \ref{thm:bridge} and \ref{thm:asymp}.
\end{proof}

\section{Conclusion}
Corollary \ref{cor:main1} provides the exponential decay required for the Kotecký-Preiss condition. Surface \#1 is complete.
