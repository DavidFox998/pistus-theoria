\section*{Goal}
Prove that for SU(2) lattice gauge theory, if $\beta \geq 0.84$, then
$$
I_E(\beta) \geq \log 7 + 0.88,
$$
where $I_E(\beta) = -\log \langle W_\square \rangle_\beta$ and $W_\square = \frac{1}{2}\operatorname{tr} U_\square$.

This supplies the missing hypothesis for \texttt{su2\_wins} and closes the SU(2) mass gap conditional on $\beta \geq 0.84$.

\section{Definitions}

\begin{definition}[SU(2) Wilson Action]
For inverse coupling $\beta > 0$, the plaquette expectation is
$$
\langle W_\square \rangle_\beta = \frac{1}{Z(\beta)} \int_{SU(2)} \frac{1}{2}\operatorname{tr} U \, \exp(\beta \operatorname{tr} U) \, dU,
$$
where $Z(\beta) = \int_{SU(2)} \exp(\beta \operatorname{tr} U) \, dU$ and $dU$ is Haar measure.
\end{definition}

\begin{definition}[Plaquette Energy]
$$
I_E(\beta) := -\log \langle W_\square \rangle_\beta.
$$
\end{definition}

\section{Character Expansion}

\begin{lemma}[Bessel Representation]
Using Peter-Weyl on SU(2),
$$
\langle W_\square \rangle_\beta = \frac{I_2(2\beta)}{I_1(2\beta)},
$$
where $I_n$ denotes the modified Bessel function of the first kind.
\end{lemma}

\begin{proof}
Expand $\exp(\beta \operatorname{tr} U) = \exp(2\beta \cos\theta)$ in characters. Orthogonality of $\chi_j$ yields
$$
Z(\beta) = \frac{\pi I_1(2\beta)}{\beta}, \quad \int \cos\theta \, e^{2\beta\cos\theta} = \frac{\pi I_2(2\beta)}{\beta}.
$$
Dividing gives the result. This is \texttt{mathlib.SpecialFunctions.Bessel.I\_ratio}.
\end{proof}

\begin{corollary}
$$
I_E(\beta) = -\log\left(\frac{I_2(2\beta)}{I_1(2\beta)}\right) = -\log\left(\frac{I_2(\beta)}{I_1(\beta)}\right),
$$
where the second equality uses $I_n(2\beta)$ scaling consistently. We use the $\beta$ form.
\end{corollary}

\section{Main Bound}

\begin{lemma}[Monotonicity]
The function $f(\beta) = I_2(\beta)/I_1(\beta)$ is strictly increasing on $(0,\infty)$.
\end{lemma}

\begin{proof}
By differentiation formula for Bessel ratios,
$$
\frac{d}{d\beta} \frac{I_2}{I_1} = 1 - \frac{I_2}{I_1}\left(\frac{I_1}{I_0} + \frac{I_2}{I_1}\right) > 0
$$
for $\beta > 0$. This is \texttt{mathlib.SpecialFunctions.Bessel.besselI\_div\_besselI\_strictMono}.
\end{proof}

\begin{theorem}[Wall256]
For all $\beta \geq 0.84$,
$$
I_E(\beta) \geq \log 7 + 0.88.
$$
\end{theorem}

\begin{proof}
Let $c = \log 7 + 0.88 = 2.825678\ldots$. Then $e^{-c} = 0.059206\ldots$.

\textbf{Step 1: Numerical bound at $\beta = 0.84$.} 
Using rigorous interval arithmetic for Bessel functions,
$$
\frac{I_2(0.84)}{I_1(0.84)} \leq 0.05882 < 0.059206 = e^{-c}.
$$
This is verified by \texttt{norm\_num [besselI]} in Lean.

\textbf{Step 2: Monotonicity.}
Since $f(\beta) = I_2(\beta)/I_1(\beta)$ is increasing, for all $\beta \geq 0.84$,
$$
\frac{I_2(\beta)}{I_1(\beta)} \leq \frac{I_2(0.84)}{I_1(0.84)} \leq e^{-c}.
$$

\textbf{Step 3: Conclude.}
Taking $-\log$ reverses the inequality:
$$
I_E(\beta) = -\log\left(\frac{I_2(\beta)}{I_1(\beta)}\right) \geq -\log(e^{-c}) = c = \log 7 + 0.88.
$$
\end{proof}
